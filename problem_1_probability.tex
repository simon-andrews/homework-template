\begin{problem}{probability}

\question{Let $A$ and $B$ be two events. Suppose that $P(A) = 0.5$ and $P(B) = 0.2$.}

\subquestion{What is the maximum possible value for $P(A \cap B)$?}
\solution
{
The maximum possible value for $P(A \cap B)$ is $0.2$. This occurs when $B$ is a subset of $A$.
}

\subquestion{What is the minimum possible value for $P(A \cap B)$?}
\solution
{
The minimum possible value for $P(A \cap B)$ is $0$. This occurs when $A$ and $B$ are disjoint sets.
}

\end{problem}
